// ✅ CONFIGURATION (YOUR REAL DATA) var SPREADSHEET_ID = "1Qmxc6YKLBKMzBQZBNMwPDynjP_MtBuiRcAZB3TfIUY0"; var QUESTION_SHEET = "Sheet 2"; // ✅ make sure this matches your questions tab var RESULT_SHEET = "Sheet 1"; // ✅ your confirmed results tab // ✅ GET RANDOM 5 QUESTIONS function doGet() { try { var sheet = SpreadsheetApp.openById(SPREADSHEET_ID) .getSheetByName(QUESTION_SHEET); var data = sheet.getDataRange().getValues(); var questions = []; for (var i = 1; i < data.length; i++) { if (!data[i][0]) continue; questions.push({ q: data[i][0], o: [ data[i][1], data[i][2], data[i][3], data[i][4] ], a: parseInt(data[i][5]) - 1 }); } // ✅ Shuffle questions.sort(() => Math.random() - 0.5); // ✅ Pick 5 var selected = questions.slice(0, 5); return ContentService .createTextOutput(JSON.stringify(selected)) .setMimeType(ContentService.MimeType.JSON); } catch (error) { return ContentService .createTextOutput(JSON.stringify({ error: error.toString() })) .setMimeType(ContentService.MimeType.JSON); } } // ✅ SAVE RESULT function doPost(e) { try { var sheet = SpreadsheetApp.openById(SPREADSHEET_ID) .getSheetByName(RESULT_SHEET); var name = e.parameter.name || ""; var score = e.parameter.score || ""; var status = e.parameter.status || ""; sheet.appendRow([ new Date(), name, score, status ]); return ContentService.createTextOutput("SUCCESS"); } catch (err) { return ContentService.createTextOutput("ERROR: " + err); } }
